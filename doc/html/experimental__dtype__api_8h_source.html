1 /*
2  * This header exports the new experimental DType API as proposed in
3  * NEPs 41 to 43. For background, please check these NEPs. Otherwise,
4  * this header also serves as documentation for the time being.
5  *
6  * Please do not hesitate to contact @seberg with questions. This is
7  * developed together with https://github.com/seberg/experimental_user_dtypes
8  * and those interested in experimenting are encouraged to contribute there.
9  *
10  * To use the functions defined in the header, call::
11  *
12  * if (import_experimental_dtype_api(version) < 0) {
13  * return NULL;
14  * }
15  *
16  * in your module init. (A version mismatch will be reported, just update
17  * to the correct one, this will alert you of possible changes.)
18  *
19  * The following lists the main symbols currently exported. Please do not
20  * hesitate to ask for help or clarification:
21  *
22  * - PyUFunc_AddLoopFromSpec:
23  *
24  * Register a new loop for a ufunc. This uses the `PyArrayMethod_Spec`
25  * which must be filled in (see in-line comments).
26  *
27  * - PyUFunc_AddPromoter:
28  *
29  * Register a new promoter for a ufunc. A promoter is a function stored
30  * in a PyCapsule (see in-line comments). It is passed the operation and
31  * requested DType signatures and can mutate it to attempt a new search
32  * for a matching loop/promoter.
33  * I.e. for Numba a promoter could even add the desired loop.
34  *
35  * - PyArrayInitDTypeMeta_FromSpec:
36  *
37  * Initialize a new DType. It must currently be a static Python C type
38  * that is declared as `PyArray_DTypeMeta` and not `PyTypeObject`.
39  * Further, it must subclass `np.dtype` and set its type to
40  * `PyArrayDTypeMeta_Type` (before calling `PyType_Read()`).
41  *
42  * - PyArray_CommonDType:
43  *
44  * Find the common-dtype ("promotion") for two DType classes. Similar
45  * to `np.result_type`, but works on the classes and not instances.
46  *
47  * - PyArray_PromoteDTypeSequence:
48  *
49  * Same as CommonDType, but works with an arbitrary number of DTypes.
50  * This function is smarter and can often return successful and unambiguous
51  * results when `common_dtype(common_dtype(dt1, dt2), dt3)` would
52  * depend on the operation order or fail. Nevertheless, DTypes should
53  * aim to ensure that their common-dtype implementation is associative
54  * and commutative! (Mainly, unsigned and signed integers are not.)
55  *
56  * For guaranteed consistent results DTypes must implement common-Dtype
57  * "transitively". If A promotes B and B promotes C, than A must generally
58  * also promote C; where "promotes" means implements the promotion.
59  * (There are some exceptions for abstract DTypes)
60  *
61  * WARNING
62  * =======
63  *
64  * By using this header, you understand that this is a fully experimental
65  * exposure. Details are expected to change, and some options may have no
66  * effect. (Please contact @seberg if you have questions!)
67  * If the exposure stops working, please file a bug report with NumPy.
68  * Further, a DType created using this API/header should still be expected
69  * to be incompatible with some functionality inside and outside of NumPy.
70  * In this case crashes must be expected. Please report any such problems
71  * so that they can be fixed before final exposure.
72  * Furthermore, expect missing checks for programming errors which the final
73  * API is expected to have.
74  *
75  * Symbols with a leading underscore are likely to not be included in the
76  * first public version, if these are central to your use-case, please let
77  * us know, so that we can reconsider.
78  *
79  * "Array-like" consumer API not yet under considerations
80  * ======================================================
81  *
82  * The new DType API is designed in a way to make it potentially useful for
83  * alternative "array-like" implementations. This will require careful
84  * exposure of details and functions and is not part of this experimental API.
85  */

343  * the header to ensure changes are taken into account. NumPy will further

349 static int
350 import_experimental_dtype_api(int version)
351 {
352  if (version != __EXPERIMENTAL_DTYPE_VERSION) {
353  PyErr_Format(PyExc_RuntimeError,
354  "DType API version %d did not match header version %d. Please "
355  "update the import statement and check for API changes.",
356  version, __EXPERIMENTAL_DTYPE_VERSION);
357  return -1;
358  }
359  if (__experimental_dtype_api_table != __uninitialized_table) {
360  /* already imported. */
361  return 0;
362  }
363 
364  PyObject *multiarray = PyImport_ImportModule("numpy.core._multiarray_umath");
365  if (multiarray == NULL) {
366  return -1;
367  }
368 
369  PyObject *api = PyObject_CallMethod(multiarray,
370  "_get_experimental_dtype_api", "i", version);
371  Py_DECREF(multiarray);
372  if (api == NULL) {
373  return -1;
374  }
375  __experimental_dtype_api_table = PyCapsule_GetPointer(api,
376  "experimental_dtype_api_table");
377  Py_DECREF(api);
378 
379  if (__experimental_dtype_api_table == NULL) {
380  __experimental_dtype_api_table = __uninitialized_table;
381  return -1;
382  }
383  return 0;
384 }
